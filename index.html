// Account balance calculation (double-entry)
function bal(id){
  const acc=LD[id];if(!acc)return 0;
  let b=(acc.opening||0)*(acc.balType==='debit'?1:-1);
  Object.values(JN).forEach(v=>{if(v.debit===id)b+=v.amount||0;if(v.credit===id)b-=v.amount||0;});
  return b;
}

window.setVT=function(type,btn){
  curVT=type;
  document.querySelectorAll('.vt').forEach(b=>b.classList.remove('on'));
  btn.classList.add('on');
  const cfg=VT_CFG[type];
  if(ge('vt-hint'))ge('vt-hint').innerHTML=cfg.hint;
  const drEl=ge('v-dr'),crEl=ge('v-cr');
  if(!drEl||!crEl)return;
  const allAcc=Object.entries(LD);
  const byType=(types)=>allAcc.filter(([,a])=>Array.isArray(types)?types.includes(a.type):a.type===types);
  const toOpts=(arr,defSel)=>arr.map(([id,a])=>``).join('');
  // Find common accounts
  const bankId=allAcc.find(([,a])=>a.type==='asset'&&(a.name.toLowerCase().includes('bank')))?.[0]||'';
  const cashId=allAcc.find(([,a])=>a.type==='asset'&&a.name.toLowerCase().includes('cash'))?.[0]||'';
  const maintId='a-maint';
  if(type==='REC'){
    drEl.innerHTML=''+toOpts(byType('asset'),bankId||cashId);
    crEl.innerHTML=''+toOpts(byType('income'),maintId);
  }else if(type==='PAY'){
    drEl.innerHTML=''+toOpts(byType('expense'),'');
    crEl.innerHTML=''+toOpts(byType('asset'),bankId||cashId);
  }else if(type==='CON'){
    drEl.innerHTML=''+toOpts(byType('asset'),bankId);
    crEl.innerHTML=''+toOpts(byType('asset'),cashId);
  }else if(type==='EXP'){
    drEl.innerHTML=''+toOpts(byType('expense'),'');
    crEl.innerHTML=''+toOpts(byType('asset'),bankId||cashId);
  }else{
    // Journal — all accounts
    drEl.innerHTML=''+toOpts(allAcc,'');
    crEl.innerHTML=''+toOpts(allAcc,'');
  }
  const c=(VC[type]||0)+1;
  if(ge('v-no'))ge('v-no').value=type+'-'+String(c).padStart(4,'0');
  ge('v-acc-hint').style.display='none';
};
